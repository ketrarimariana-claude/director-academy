🗄️ Supabase — Schema SQL
Ruleaza acest SQL in Supabase → SQL Editor pentru a crea tabelele necesare
-- Tabel progres per utilizator
CREATE TABLE IF NOT EXISTS progress (
  id BIGSERIAL PRIMARY KEY,
  user_email TEXT NOT NULL,
  question_id TEXT NOT NULL,
  correct BOOLEAN,
  times INTEGER DEFAULT 1,
  last_date TEXT,
  UNIQUE(user_email, question_id)
);

-- Tabel clasament
CREATE TABLE IF NOT EXISTS leaderboard (
  user_email TEXT PRIMARY KEY,
  user_name TEXT,
  school TEXT,
  score INTEGER DEFAULT 0,
  studied INTEGER DEFAULT 0,
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Acces public de citire clasament
ALTER TABLE leaderboard ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read" ON leaderboard FOR SELECT USING (true);
CREATE POLICY "user write" ON leaderboard FOR ALL USING (auth.email() = user_email);
